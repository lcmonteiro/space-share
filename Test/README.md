# Coding Style

```cpp
class SObject {
public:
    /**
     * ------------------------------------------------------------------------
     * helper types
     * ------------------------------------------------------------------------
     */
    using Type1 = SModule::Command;
    using Type2 = SModule::Command;
    using Typen = SModule::Command;
    /**
     * ------------------------------------------------------------------------
     * interface functions
     * ------------------------------------------------------------------------
     * verb_object
     * ----------------------------------------------------
     */
    Type1 verb_object();
    /**
     * ----------------------------------------------------
     * verb_object
     * @obj 
     * ----------------------------------------------------
     */
    Type2 verb_object(Type2 obj);
private:
    /**
     * ------------------------------------------------------------------------
     * implementation functions
     * ------------------------------------------------------------------------
     * verb
     * ----------------------------------------------------
     */
    void __verb();
    /**
     * ------------------------------------------------------------------------
     * Variables
     * ------------------------------------------------------------------------
     * object 2
     * ----------------------------------------------------
     */
    Type2 __obj1;
    /**
     * ----------------------------------------------------
     * object 2
     * ----------------------------------------------------
     */
    Type2 __obj2;
    /**
     * ----------------------------------------------------
     * object n
     * ----------------------------------------------------
     */
    Type2 __objn;
};
```
